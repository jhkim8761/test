create table TEST.PERSON
(
PERSON_ID    varchar2(15) not NULL primary KEY,
PERSON_NAME   varchar2(15)
);

COMMIT;

create table TEST.PRODUCT
(
PRODUCT_ID    varchar2(15) not NULL primary KEY,
PRODUCT_NAME   varchar2(15)
);

COMMIT;


create table TEST.ORDERTABLE
(
ORDER_ID	VARCHAR2(15)	NOT NULL primary KEY,
PERSON_ID   varchar2(15)	REFERENCES TEST.PERSON(PERSON_ID),
PRODUCT_ID  varchar2(15)	REFERENCES TEST.PRODUCT(PRODUCT_ID),
ORDER_DATE	DATE
);
